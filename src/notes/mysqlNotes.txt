CREATE DATABASE GeeksForGeeks # creates an empty DATABASE
SHOW DATABASES;
USE GeeksForGeeks; # switches to that sql DATABASE
DROP DATABASE GeeksForGeeks; # deletes

# Creates a table with some given keys:
CREATE TABLE Customer(
    CustomerID INT PRIMARY KEY,
    CustomerName VARCHAR(50),
    LastName VARCHAR(50),
    Country VARCHAR(50),
    Age INT CHECK (Age >= 0 AND Age <= 99),
    Phone int(10)
);
